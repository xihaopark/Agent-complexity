library(DESeq2)

# Load
counts <- read.table("input/counts.tsv", header=TRUE, row.names="gene_id")
coldata <- read.table("input/coldata.tsv", header=TRUE, row.names="sample")

# DESeq2
dds <- DESeqDataSetFromMatrix(countData=counts, colData=coldata, design=~condition)
dds <- dds[rowSums(counts(dds)) >= 10, ]
dds <- DESeq(dds)

# Results
res <- results(dds, name="condition_treated_vs_control")

# Shrink LFC - use ashr (general purpose, modern default)
res <- lfcShrink(dds, coef=2, type="ashr", res=res)

# Alternative shrinkage types to consider:

# - type="apeglm": For small samples (if available)

# Output
write.csv(as.data.frame(res), "output/shrunk_de.csv")
