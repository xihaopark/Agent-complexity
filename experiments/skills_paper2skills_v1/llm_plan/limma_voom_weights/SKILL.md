```r
library(limma)
library(edgeR)

# Load
counts <- read.table("input/counts.tsv", header=TRUE, row.names="gene_id")
coldata <- read.table("input/coldata.tsv", header=TRUE, row.names="sample")

# DGEList
dge <- DGEList(counts=counts, group=coldata$condition)
dge <- calcNormFactors(dge, method="TMM")

# Design
design <- model.matrix(~ condition, data=coldata)

# voom
v <- voom(dge, design, plot=FALSE)

# Linear model
fit <- lmFit(v, design)
fit <- eBayes(fit)

# Results
res <- topTable(fit, coef=2, number=Inf, sort.by="none")

# Output
write.csv(res, "output/de_results_weighted.csv")
```

## Notes

- Standard voom pipeline assumes good sample quality
- If samples have quality issues, consider voomWithQualityWeights()
- Output column names differ from DESeq2 (logFC vs log2FoldChange)
