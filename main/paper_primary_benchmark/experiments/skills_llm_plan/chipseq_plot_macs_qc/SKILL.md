## Method
The task involves computing summary statistics for ChIP-seq peak data generated by MACS2. The objective is to calculate descriptive statistics (Min, 1st Quartile, Median, Mean, 3rd Quartile, Max) for specific columns (`fold`, `-log10(qvalue)`, `-log10(pvalue)`, and peak `length`) from narrowPeak files. Additionally, the total number of peaks (`num_peaks`) for each sample should be included. The results should be compiled into a single table with columns for each statistic, the measure, and the sample identifier.

- Columns to summarize: `fold`, `-log10(qvalue)`, `-log10(pvalue)`, `length`

# Load necessary library
library(dplyr)

# Define a function to compute summary statistics
compute_summary <- function(file_path, sample_name) {
  data <- read.table(file_path, header = FALSE, sep = "\t")
  colnames(data) <- c("chr", "start", "end", "name", "score", "strand", "fold", "pvalue", "qvalue", "peak")
  
  # Calculate peak length
  data$length <- data$end - data$start
  
  # Compute summary statistics
  summary_stats <- data %>%
    summarise(
      num_peaks = n(),
      fold_min = min(fold),
      fold_1st_qu = quantile(fold, 0.25),
      fold_median = median(fold),
      fold_mean = mean(fold),
      fold_3rd_qu = quantile(fold, 0.75),
      fold_max = max(fold),
      qvalue_min = min(-log10(qvalue)),
      qvalue_1st_qu = quantile(-log10(qvalue), 0.25),
      qvalue_median = median(-log10(qvalue)),
      qvalue_mean = mean(-log10(qvalue)),
      qvalue_3rd_qu = quantile(-log10(qvalue), 0.75),
      qvalue_max = max(-log10(qvalue)),
      pvalue_min = min(-log10(pvalue)),
      pvalue_1st_qu = quantile(-log10(pvalue), 0.25),
      pvalue_median = median(-log10(pvalue)),
      pvalue_mean = mean(-log10(pvalue)),
      pvalue_3rd_qu = quantile(-log10(pvalue), 0.75),
      pvalue_max = max(-log10(pvalue)),
      length_min = min(length),
      length_1st_qu = quantile(length, 0.25),
      length_median = median(length),
      length_mean = mean(length),
      length_3rd_qu = quantile(length, 0.75),
      length_max = max(length)
    ) %>%
    mutate(sample = sample_name)
  
  return(summary_stats)
}

# Compute summaries for each sample
summary_A <- compute_summary("input/sampleA_peaks.narrowPeak", "sampleA")
summary_B <- compute_summary("input/sampleB_peaks.narrowPeak", "sampleB")
summary_C <- compute_summary("input/sampleC_peaks.narrowPeak", "sampleC")

# Combine summaries
summary.dat <- bind_rows(summary_A, summary_B, summary_C)

# Write to output file
write.table(summary.dat, "output/macs_qc_summary.tsv", sep = "\t", row.names = FALSE, col.names = TRUE, quote = FALSE)
