The task involves filtering GFF files generated by histoneHMM based on the `avg_posterior` attribute. - Filtering threshold: `avg_posterior >= 0.5`

## Commands / Code Snippets
```r
library(rtracklayer)

# Function to process each sample
process_sample <- function(input_file, output_gff, output_bed) {
  # Import GFF file
  gff_data <- import.gff(input_file)
  
  # Filter based on avg_posterior
  filtered_data <- gff_data[gff_data$avg_posterior >= 0.5, ]
  
  # Export to GFF3
  export.gff3(filtered_data, output_gff)
  
  # Set score and export to BED
  filtered_data$score <- as.numeric(filtered_data$avg_posterior)
  export.bed(filtered_data, output_bed)
}

# Process each sample
process_sample("input/sampleA.filtered.histoneHMM-regions.gff", 
               "output/sampleA_avgp0.5.gff", 
               "output/sampleA_avgp0.5.bed")

process_sample("input/sampleB.filtered.histoneHMM-regions.gff", 
               "output/sampleB_avgp0.5.gff", 
               "output/sampleB_avgp0.5.bed")
```
